#define RELE1 14
#define RELE2 27
#define RELE3 26

void setup() {
  Serial.begin(115200);

  pinMode(RELE1, OUTPUT);
  pinMode(RELE2, OUTPUT);
  pinMode(RELE3, OUTPUT);

  // IMPORTANTE: depende del módulo rele
  // Si tu relé es activo en LOW, los apagamos así:
  digitalWrite(RELE1, HIGH);
  digitalWrite(RELE2, HIGH);
  digitalWrite(RELE3, HIGH);

  Serial.println("Iniciando prueba de reles...");
}

void loop() {
  // Encender todos
  Serial.println("ENCENDIENDO RELES");
  digitalWrite(RELE1, LOW);
  digitalWrite(RELE2, LOW);
  digitalWrite(RELE3, LOW);
  delay(2000);

  // Apagar todos
  Serial.println("APAGANDO RELES");
  digitalWrite(RELE1, HIGH);
  digitalWrite(RELE2, HIGH);
  digitalWrite(RELE3, HIGH);
  delay(2000);
}
