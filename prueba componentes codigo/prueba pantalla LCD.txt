Pantalla LCD
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// Dirección I2C típica del módulo LCD (usa 0x27 o 0x3F)
LiquidCrystal_I2C lcd(0x27, 16, 2);  // (Dirección, columnas, filas)

void setup() {
  Wire.begin(21, 22);  // SDA=21, SCL=22 para ESP32
  lcd.init();           // Inicializa el LCD
  lcd.backlight();      // Enciende la luz de fondo
  
  lcd.setCursor(0, 0);
  lcd.print("Hola ESP32!");
  lcd.setCursor(0, 1);
  lcd.print("LCD funcionando");
}

void loop() {
  // Ejemplo de texto animado
  lcd.setCursor(0, 1);
  lcd.print("Contador: ");
  lcd.print(millis() / 1000); // Segundos desde inicio
  delay(1000);
}
